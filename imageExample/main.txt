function main() {
	clear();
	append("This is an example of displaying images in mods");

	addSprite("art.png");
	addChoice("Next", next);
}

function next() {
	clear();
	append("Here click below to move this image around");
	
	var image = addSprite("art.png");
	image.y = 200;
	
	addChoice("Move", function() {
		image.x = rndInt(0, 800);
		image.y = rndInt(0, 600);
	});

	addChoice("Leave", modExit);
}

main();
